Generics
// Generic function
function identity<T>(x: T): T {
  return x;
}
identity<string>("hi");
identity(42); // inferred

// Generic class
class Box<T> {
  constructor(public value: T) {}
  map<U>(f: (v: T) => U): Box<U> {
    return new Box(f(this.value));
  }
}

// Constraints
function getLen<T extends { length: number }>
  (x: T) { return x.length; }
// Multiple type parameters
function pair<A, B>(a: A, b: B):
  [A, B] { return [a, b]; }

// keyof Constraint
function getVal<T, K extends keyof T>
  (obj: T, key: K): T[K] {
  return obj[key];
}

// Default type
type List<T = string> = T[];

// Conditional Type
type IsArray<T> =
  T extends any[] ? "yes" : "no";

// Infer
type UnwrapArray<T> =
  T extends (infer U)[] ? U : T;

class Service {
  // Access Modifier im Ctor (Shorthand)
  constructor(
    private readonly url: string,
    protected name: string,
    public version = 1
  ) {}

  // Private Field (ES#)
  #secret = "hidden";

  // Abstract Method
  abstract run(): void;

  // Static
  static instance: Service;
  static create(): Service { ... }

  // Getter / Setter
  get fullUrl(): string {
    return `${this.url}/v${this.version}`;
  }
}

Decorators (TS 5.x)
// Classes-Decorator
function sealed(target: typeof MyClass) {
  Object.seal(target);
}
@sealed
class MyClass { }

// Method decorator
function log(target: any, ctx: ClassMethodDecoratorContext) {
  return function(...args: any[]) {
    console.log(`Calling ${String(ctx.name)}`);
    return (this as any)[ctx.name](...args);
  };
}

// Property Decorator
function readonly(t: any, ctx: ClassFieldDecoratorContext) {
  return () => { Object.freeze(t); };
}

// Usage
class Greeter {
  @log
  greet(name: string) { return name; }
}

Advanced Types: Mapped, Conditional, Template Literals & Modules
Mapped Types
// All properties optional
type Optional<T> = {
  [P in keyof T]?: T[P]
};

// Filter properties via as
type OnlyStrings<T> = {
  [K in keyof T as
    T[K] extends string ? K : never]: string
};

// Getters for all props
type Getters<T> = {
  [K in keyof T as
    `get${Capitalize<string & K>}`
  ]: () => T[K]
};
Conditional & Infer
// Flatten array
type Flat<T> =
  T extends (infer U)[] ? U : T;

// Promise unwrap
type Unpack<T> =
  T extends Promise<infer V> ? V : T;

// Deep Partial
type DeepPartial<T> = {
  [P in keyof T]?:
    T[P] extends object
      ? DeepPartial<T[P]>
      : T[P]
};

// IsNever
type IsNever<T> =
  [T] extends [never] ? true : false;

Assertion & Satisfies
// Type Assertion
const x = val as string;
const x = <string>val;
const x = val as unknown as string;

// Non-null Assertion
el!.addEventListener("click", fn);

// satisfies (TS 4.9)
const config = {
  port: 8080,
  host: "localhost"
} satisfies Config;
// preserves exact type!

// const assertion
const arr = [1, 2] as const;
// readonly [1, 2]

// Assertion Functions
function assert(v: unknown):
  asserts v {
  if (!v) throw new Error();
}
