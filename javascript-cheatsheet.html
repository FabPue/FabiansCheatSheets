Classes (ES6+)
class Animal {
  // Private Field (ES2022)
  #name;
  static count = 0;

  constructor(name) {
    this.#name = name;
    Animal.count++;
  }

  get name() { return this.#name; }
  set name(v) { this.#name = v; }

  speak() { return `${this.#name} says`; }
  static create(n) { return new Animal(n); }
}

class Dog extends Animal {
  #breed;
  constructor(name, breed) {
    super(name);
    this.#breed = breed;
  }
  speak() { return super.speak() + " Woof!"; }
}

const d = new Dog("Rex", "Labrador");

// Generator Function
function* range(start, end) {
  for (let i=start; i<end; i++) yield i;
}
const gen = range(0, 5);
gen.next();  // { value: 0, done: false }
[...range(0,5)]  // [0,1,2,3,4]

// Iterable Protocol
const iterable = {
  [Symbol.iterator]() {
    let n = 0;
    return {
      next() {
        return n < 3
          ? { value: n++, done: false }
          : { done: true };
      }
    };
  }
};
