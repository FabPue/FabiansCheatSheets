Pointer & Address
// Pointer declaration
int *ptr;           // Pointer to int
int *p = NULL;     // Null pointer
char *s = "Hi";    // String pointer
void *vp;           // generic pointer

// Address & dereference
int x = 42;
int *p = &x;       // address of x
*p = 100;          // x = 100
printf("%d\n", *p); // 100

// Pointer arithmetic
p++;// next int address
p += 3;
*(p + 1)           // next element

// Pointer to array
int arr[5] = {1,2,3,4,5};
int *ap = arr;     // arr = &arr[0]
ap[2] == *(ap+2); // equivalent
// Const pointer variants
const int *p;      // Ptr to const int
int * const p;     // const ptr to int
const int * const p; // both const

// Double pointer
int **pp = &p;
**pp = 42;

// Function pointer
int (*func_ptr)(int, int);
func_ptr = &add;
int result = func_ptr(3, 4);

// Typedef for function pointer
typedef int (*MathFunc)(int, int);
MathFunc fn = add;

// Pointer comparison
if (ptr == NULL) { }
if (ptr != NULL) { }
p1 - p2; // ptrdiff_t

int sum(int arr[], int n); // equivalent

// Variadic function
#include <stdarg.h>
int sum(int n, ...) {
  va_list ap;
  va_start(ap, n);
  for(int i=0;i<n;i++) va_arg(ap,int);
  va_end(ap);
}

// Inline function (C99)

// struct
struct Point {
  int x;
  int y;
};
struct Point p = { 3, 4 };
struct Point p = { .x=3, .y=4 }; // designated init
p.x = 10;

// Pointer to struct
struct Point *pp = &p;
pp->x = 99;    // (*pp).x = 99

// typedef for struct
typedef struct {
  char name[50];
  int  age;
} Person;
Person anna = { "Anna", 30 };

// Struct array
Person people[100];
people[0].age = 25;
// union (shared memory)
union Data {
  int    i;
  float  f;
  char   str[20];
};
union Data d;
d.i = 10;
d.f = 3.14; // overwrites i!

// enum
enum Color { RED=0, GREEN, BLUE };
enum Color c = GREEN;

// Bit fields in struct
struct Flags {
  unsigned read    : 1;
  unsigned write   : 1;
  unsigned execute : 1;
  unsigned padding : 5;
};

// Nested structs
struct Rect {
  struct Point top_left;
  struct Point bottom_right;

Linked List Pattern
typedef struct Node {
  int data;
  struct Node *next;
} Node;

Node* create(int val) {
  Node *n = malloc(sizeof(Node));
  n->data = val;
  n->next = NULL;
  return n;
}

// Traverse
Node *cur = head;
while (cur) {
  printf("%d ", cur->data);
  cur = cur->next;
}

// Free everything
while (head) {
  Node *tmp = head;
  head = head->next;
  free(tmp);
}
